Section Internal_Adjunction.
  Context {B : bicat}.

  (** ** Definitions of internal adjunctions *)

  (** *** Data & laws for left adjoints *)
  Definition left_adjoint_data
             {a b : B}
             (f : a --> b)
    : UU
    := ‚àë (g : b --> a),
       identity a ==> f ¬∑ g
       √ó
       g ¬∑ f ==> identity b.

  Definition left_adjoint_right_adjoint
             {a b : B}
             {f : a --> b}
             (Œ±d : left_adjoint_data f)
    : b --> a
    := pr1 Œ±d.

  Definition left_adjoint_unit
             {a b : B}
             {f : a --> b}
             (Œ±d : left_adjoint_data f)
    : identity a ==> f ¬∑ left_adjoint_right_adjoint Œ±d
    := pr12 Œ±d.

  Definition left_adjoint_counit
             {a b : B}
             {f : a --> b}
             (Œ±d : left_adjoint_data f)
    : left_adjoint_right_adjoint Œ±d ¬∑ f ==> identity b
    := pr22 Œ±d.

  Definition left_adjoint_axioms
             {a b : B}
             {f : a --> b}
             (Œ±d : left_adjoint_data f)
    : UU
    := let g := left_adjoint_right_adjoint Œ±d in
       let Œ∑ := left_adjoint_unit Œ±d in
       let Œµ := left_adjoint_counit Œ±d in
       linvunitor f ‚Ä¢ (Œ∑ ‚ñπ f) ‚Ä¢ rassociator _ _ _ ‚Ä¢ (f ‚óÉ Œµ) ‚Ä¢ runitor f = id2 f
       √ó
       rinvunitor g ‚Ä¢ (g ‚óÉ Œ∑) ‚Ä¢ lassociator _ _ _ ‚Ä¢ (Œµ ‚ñπ g) ‚Ä¢ lunitor g = id2 g.

  Definition left_adjoint
             {a b : B}
             (f : a --> b)
    : UU
    := ‚àë (Œ±d : left_adjoint_data f), left_adjoint_axioms Œ±d.

  Coercion data_of_left_adjoint
           {a b : B}
           {f : a --> b}
           (Œ± : left_adjoint f)
    : left_adjoint_data f
    := pr1 Œ±.

  Coercion axioms_of_left_adjoint
           {a b : B}
           {f : a --> b}
           (Œ± : left_adjoint f)
    : left_adjoint_axioms Œ±
    := pr2 Œ±.

  (** Data and laws for right adjoints *)
  Definition internal_right_adj_data
             {a b : B}
             (g : b --> a)
    : UU
    := ‚àë (f : a --> b),
       identity a ==> f ¬∑ g
       √ó
       g ¬∑ f ==> identity b.

  Definition internal_right_adj_left_adjoint
             {a b : B}
             {g : b --> a}
             (Œ±d : internal_right_adj_data g)
    : a --> b
    := pr1 Œ±d.

  Definition internal_right_adj_unit
             {a b : B}
             {g : b --> a}
             (Œ±d : internal_right_adj_data g)
    : identity a ==> internal_right_adj_left_adjoint Œ±d ¬∑ g
    := pr12 Œ±d.

  Definition internal_right_adj_counit
             {a b : B}
             {g : b --> a}
             (Œ±d : internal_right_adj_data g)
    : g ¬∑ internal_right_adj_left_adjoint Œ±d ==> identity b
    := pr22 Œ±d.

  Definition internal_right_adj_axioms
             {a b : B}
             {g : b --> a}
             (Œ±d : internal_right_adj_data g)
    : UU
    := let f := internal_right_adj_left_adjoint Œ±d in
       let Œ∑ := internal_right_adj_unit Œ±d in
       let Œµ := internal_right_adj_counit Œ±d in
       linvunitor f ‚Ä¢ (Œ∑ ‚ñπ f) ‚Ä¢ rassociator _ _ _ ‚Ä¢ (f ‚óÉ Œµ) ‚Ä¢ runitor f = id2 f
       √ó
       rinvunitor g ‚Ä¢ (g ‚óÉ Œ∑) ‚Ä¢ lassociator _ _ _ ‚Ä¢ (Œµ ‚ñπ g) ‚Ä¢ lunitor g = id2 g.

  Definition internal_right_adj
             {a b : B}
             (g : b --> a)
    : UU
    := ‚àë (Œ±d : internal_right_adj_data g), internal_right_adj_axioms Œ±d.

  Coercion data_of_internal_right_adj
           {a b : B}
           {g : b --> a}
           (Œ± : internal_right_adj g)
    : internal_right_adj_data g
    := pr1 Œ±.

  Coercion axioms_of_internal_right_adj
           {a b : B}
           {g : b --> a}
           (Œ± : internal_right_adj g)
    : internal_right_adj_axioms Œ±
    := pr2 Œ±.

  (** *** Laws for equivalences *)
  Definition left_equivalence_axioms
             {a b : B}
             {f : a --> b}
             (Œ±d : left_adjoint_data f)
    : UU
    := is_invertible_2cell (left_adjoint_unit Œ±d)
       √ó
       is_invertible_2cell (left_adjoint_counit Œ±d).

  Definition left_equivalence
             {a b : B}
             (f : a --> b)
    :UU
    := ‚àë (Œ±d : left_adjoint_data f),
       left_equivalence_axioms Œ±d.

  Coercion data_of_left_equivalence
           {a b : B}
           {f : a --> b}
           (Œ±e : left_equivalence f)

    := pr1 Œ±e.

  Coercion axioms_of_left_equivalence
           {a b : B}
           {f : a --> b}
           (Œ±e : left_equivalence f)
    : left_equivalence_axioms Œ±e
    := pr2 Œ±e.

  Definition left_adjoint_equivalence
             {a b : B}
             (f : a --> b)

    := ‚àë (Œ±d : left_adjoint_data f),
       left_adjoint_axioms Œ±d

  (** *** Packaged *)
  Definition adjunction (a b : B) : UU
    := ‚àë (f : a --> b), left_adjoint f.

  Coercion arrow_of_adjunction
           {a b : B}
           (f : adjunction a b)
    : a --> b
    := pr1 f.

  Coercion left_adjoint_of_adjunction
           {a b : B}
           (f : adjunction a b)
    : left_adjoint f
    := pr2 f.

  Definition adjoint_equivalence
             (a b : B)
    : UU
    := ‚àë (f : a --> b), left_adjoint_equivalence f.

  Coercion adjunction_of_adjoint_equivalence
           {a b : B}
           (f : adjoint_equivalence a b)

    := (pr1 f,,left_adjoint_of_left_adjoint_equivalence (pr2 f)).

  Coercion left_adjoint_equivalence_of_adjoint_equivalence
           {a b : B}
           (f : adjoint_equivalence a b)
    : left_adjoint_equivalence f
    := pr2 f.

  Definition internal_right_adjoint
             {a b : B}
             (f : adjunction a b)
    : b --> a
    := left_adjoint_right_adjoint f.

  Definition internal_triangle1
